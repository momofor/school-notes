\section{Derivatives}
\definition{
	A derivative Is the rate of change of a function under a finitely small value noted $dx$.

	For a function $f(x)$ defined in $\RR$; It's derivative is written  under the form:
	\[ \frac{df}{dx}= \lim_{x\to 0} \frac{f(dx+x)-f(x)}{dx} \]

	Which gives us the formal definition of derivatives; They are the rate of change of a function
	$df$ after we move it by a little amount $dx$.
}
\property{
	Derivatives have many rules most importantly
	\begin{itemize}
		\item $\frac{d(x^n)}{dx} = \lim_{x\to 0} n*x^{n-1}.$

	\end{itemize}
}
